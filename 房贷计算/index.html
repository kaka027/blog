    首付

    import pandas as pd
price = int(input('总价/万：'))
mianji = int(input('面积 1：小于90平 2：大于90平：'))
qishui = (price * 0.01 if mianji == 1 else price * 0.015)
weiyi = int(input('是否满五唯一；1：是；2:不是： '))
geshui = (0 if weiyi == 1 else price * 0.01)
zzs = int(input(r'1:无增值税；2：不满2年/满二不唯一'))
zengzhishui = (0 if zzs == 1 else price*0.05)
zhongjie = price * 0.015
shoufu = price * 0.3
sum_shoufu = shoufu + qishui + geshui + zengzhishui + zhongjie
df = pd.DataFrame({'总价': price, '契税': qishui, '个税': geshui, '中介费': zhongjie,
                  '增值税': zengzhishui, '首付': shoufu, '首付总额': sum_shoufu}, index=[1, ])
df
