    大小写转换演示实例

    # string.capitalize()
var1 = "hello python"
var2 = "hello PYTHON"
var3 = "123hello PYTHON"
print(var1.capitalize())
print(var2.capitalize())
print(var3.capitalize())
>>>
Hello python
Hello python
123hello python

# string.title()
var1 = "this is good day"
var2 = "this is 2b2b2b and g3g3g3"
print(var1.title())
print(var2.title())
>>>
This Is Good Day
This Is 2B2B2B And G3G3G3

# string.lower()
var1 = "HelloPYTHon"
print(var1.lower())
>>>
hellopython

# string.upper()
var1 = "hello python"
print(var1.upper())
>>>
HELLO PYTHON

# string.swapcase()
var1 = "hello PYTHON"
print(var1.swapcase())
>>>
HELLO python
