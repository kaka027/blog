// 创建tr标签
var tr = document.createElement("tr");

for (var key in info) {
    //循环获取对应键的值，每获取一次创建td标签并写入值
    var text = info[key];
    var td = document.createElement("td");
    td.innerText = text;
    // 将td加入到tr标签内
    tr.appendChild(td);

    infoList = [
    {id: 809, name: "小明1", age: 19},
    {id: 810, name: "小明2", age: 19},
    {id: 811, name: "小明3", age: 19},

for (var idx in infoList ){
    info = infoList[idx];
    // 创建tr标签
    var tr = document.createElement("tr");
    for (var key in info) {
        //循环获取对应键的值，每获取一次创建td标签并写入值
        var text = info[key];
        var td = document.createElement("td");
        td.innerText = text;
        // 将td加入到tr标签内
        tr.appendChild(td);
    }
    // 根据ID获取HTML内的标签
    var bodyTag = document.getElementById("body");
    // 将tr标签加入到ID为body的标签内
    bodyTag.appendChild(tr)
}

    9.1 事件的绑定（案例）

    • HTML
    <!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
    <input type="text" placeholder="请输入内容" id="userTxt">
    <input type="button" value="点击添加" onclick="addInfo()"> <!--onclick表示单击;ondblclick表示双击-->
    <ul id="itemTag"></ul>

    <script src="static/js/2.事件的绑定.js"></script>
</body>
</html>
    • javascript
    function addInfo(){
    // 1. 通过ID找到用户输入的标签
    var txtTag = document.getElementById("userTxt");

    //2.获取标签内输入的内容
    var newString = txtTag.value;

    //3.判断输入框内输入内容是否为空
    if (newString.length>0) {
        //4. 创建li标签
        var li = document.createElement("li");
        //5. 给li标签内填充内容
        li.innerText = newString;
        //6.把li标签添加到ul标签中
        var parentTag = document.getElementById("itemTag");
        parentTag.appendChild(li)
        //7.将输入的input框内容清空
        txtTag.value = ""
    }else {
        //如果输入为空，提醒
        alert("输入不能为空")
    }
}
